## Usage

```js
var snapdragonCheerio = require('{%= name %}');
var ast = snapdragonCheerio('<p>Foo</p>');

console.log(ast);
// {
//   type: 'string',
//   nodes: [
//     {type: 'bos', val: ''}, // beginning of string
//     {type: 'p', attribs: {}, 
//       nodes: [
//         {type: 'p.open', val: ''},
//         {type: 'text', val: 'Foo'},
//         {type: 'p.close', val: ''}
//       ]
//     },
//     {type: 'eos', val: ''}  // end of string
//   ]
// }
```
